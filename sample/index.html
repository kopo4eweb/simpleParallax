<script src="jquery.simpleparallax.js"></script>

Default

$(document).ready(function () {
    $('#example_1').simpleParallax();
});

default options plugin

parallaxRun: 'down'

$(document).ready(function () {
    $('#example_2').simpleParallax({ parallaxRun: 'down' });
});

parallaxRun: 'down'

parallaxRun: 'up', parallaxZoom: '+'

$(document).ready(function () {
    $('#example_2').simpleParallax({ parallaxRun: 'up', parallaxZoom: '+' });
});

parallaxRun: 'up', parallaxZoom: '+'

parallaxArea: 80

$(document).ready(function () {
    $('#example_4').simpleParallax({ parallaxArea:80 });
});

parallaxArea: 80
